\section*{Key bindings (FZF)}
\begin{tabular}{@{}l@{\hspace{1em}}l@{}}

\kbd{Ctrl-R}       & reverse history search \\
\kbd{Alt-C}        & cd (fzf) \\
\kbd{Alt-Shift-C}  & cd incl. hidden \\
\kbd{Ctrl-G}       & open file \\
\kbd{Ctrl-O}       & open in editor \\
\kbd{Tab}          & fzf completion (if \cmd{FZF\_COMPLETE}) \\
\end{tabular}

\section*{fd (find)}
\begin{kvtable}
fd \textit{pat}            & find by name \\
fd -e py \textit{pat}      & filter by extension \\
fd -t d \textit{pat}       & dirs only (\cmd{-t f} files) \\
fd -H                      & include hidden \\
fd -I                      & ignore \cmd{.gitignore} \\
fd \textit{pat} \textit{path}  & search under path \\
fd -x \textit{cmd} \{\}    & run cmd per match \\
\end{kvtable}

\section*{bat (cat)}
\begin{kvtable}
bat \textit{file}              & view w/ syntax \\
bat -p \textit{file}           & plain (pipe-safe) \\
bat -n \textit{file}           & line numbers only \\
bat -r 10:30 \textit{file}     & line range \\
bat --diff \textit{file}       & only changed hunks \\
bat -l rust \textit{file}      & force language \\
\end{kvtable}
