\section*{Navigation}
\begin{tabular}{@{}>{\ttfamily}l@{\hspace{1em}}l@{}}
code        & fzf-pick under \cmd{\textasciitilde/Code} \\
code \textit{name} & cd \cmd{\textasciitilde/Code/\textit{name}} \\
z \textit{query}   & smart cd (zoxide) \\
nvm use \textit{v} & activate Node version \\
nvm install \textit{v} & install Node version \\
nvm ls      & list installed versions \\
fisher update  & update plugins \\
fisher list    & list installed plugins \\
\end{tabular}
